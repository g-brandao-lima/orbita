def _make_snapshot(db, route_group, origin="GRU", destination="GIG",
                   dep_date=date(2026, 6, 15), ret_date=date(2026, 6, 22),
                   price=1500.0, classification=None, booking_classes=None):
    """Creates and persists a FlightSnapshot with optional BookingClassSnapshot list."""
    snapshot = FlightSnapshot(
        route_group_id=route_group.id, origin=origin, destination=destination,
        departure_date=dep_date, return_date=ret_date, price=price,
        currency="BRL", airline="LA", price_classification=classification,
    )
    db.add(snapshot)
    db.flush()  # get snapshot.id
    if booking_classes:
        for bc in booking_classes:
            db.add(BookingClassSnapshot(
                flight_snapshot_id=snapshot.id,
                class_code=bc["class_code"],
                seats_available=bc["seats"],
                segment_direction=bc.get("direction", "OUTBOUND"),
            ))
    db.flush()
    # Reload to get relationships populated
    db.refresh(snapshot)
    return snapshot

def _make_route_group(db, name="Test Group"):

        duration_days=7, travel_start=date(2026, 6, 1), travel_end=date(2026, 7, 31),
    )
    db.add(group)
    db.flush()
    return group
